# --- Logavimo Konfigūracija (Prieš bet kokius kitus importus, kad užtikrintume, jog veikia nuo pat pradžių) ---
def setup_logging():
    """Konfigūruoja išsamų logavimą visai sistemai."""
    log_filename = f"{Config.LOG_FILE_PREFIX}_{int(time.time())}.log"
    
    # Sukuriamas pagrindinis logger'is
    logger = logging.getLogger()
    logger.setLevel(getattr(logging, Config.LOG_LEVEL.upper()))

    # Išvalo esamus handler'ius, kad nebūtų dubliavimo, jei funkcija kviečiama kelis kartus
    if logger.handlers:
        for handler in logger.handlers:
            logger.removeHandler(handler)

    # Failo handler'is
    file_handler = logging.FileHandler(log_filename, mode='w')
    file_handler.setFormatter(logging.Formatter(
        '%(asctime)s - %(name)s - %(levelname)s - %(message)s'
    ))
    logger.addHandler(file_handler)

    # Konsolės handler'is
    console_handler = logging.StreamHandler()
    console_handler.setFormatter(logging.Formatter(
        '%(asctime)s - %(levelname)s - %(message)s'
    ))
    logger.addHandler(console_handler)
    
    logging.info(f"Logavimas nustatytas. Logai rašomi į: {log_filename}")

setup_logging() # Iškart nustatome logavimą

# --- Pagalbinės Funkcijos ir Stebėsenos Įrankiai ---
def get_current_vram_usage_mb() -> float:
    """Grąžina dabartinį VRAM naudojimą MB, jei CUDA prieinama."""
    if torch.cuda.is_available():
        torch.cuda.empty_cache() # Pabandykime išvalyti, kad gautume tikslesnį rodmenį
        allocated_bytes = torch.cuda.memory_allocated()
        cached_bytes = torch.cuda.memory_reserved()
        logging.debug(f"VRAM Allocated: {allocated_bytes / (1024**2):.2f} MB, Cached: {cached_bytes / (1024**2):.2f} MB")
        return allocated_bytes / (1024**2)
    else:
        logging.warning("CUDA neprieinama. VRAM naudojimas negali būti stebimas realiu laiku. Grąžinama 0.0 MB.")
        return 0.0

def check_vram_limit(operation_name: str, genome_id: str):
    """Patikrina VRAM naudojimą ir logina."""
    current_vram = get_current_vram_usage_mb()
    if current_vram > Config.VRAM_LIMIT_MB:
        logging.error(f"[Genome {genome_id}] VRAM limitas viršytas! Operacija: {operation_name}. Naudojama: {current_vram:.2f} MB, Limitas: {Config.VRAM_LIMIT_MB} MB.")
        return False # Grąžiname False, kad fitneso funkcija galėtų tai įvertinti
    else:
        logging.info(f"[Genome {genome_id}] VRAM naudojimas po '{operation_name}': {current_vram:.2f} MB / {Config.VRAM_LIMIT_MB} MB. Viskas gerai.")
        return True # Grąžiname True, jei viskas gerai

# --- Genomo Apibrėžimas ---
class Genome:
    """
    Genomas – visos GeneticArchitect sistemos "brėžinys".
    Visi sistemos parametrai, kuriuos evoliucionuoja GA, yra čia.
    """
    def __init__(self, genome_id: str = None):
        self.id = genome_id if genome_id else str(uuid.uuid4())
        self.fitness: float = -float('inf') # Kuo didesnis, tuo geriau
        self.vram_used_mb: float = 0.0
        self.hnn_mse_loss: float = float('inf')
        self.compression_ratio: float = 0.0 # Pvz., 100x = 100.0

        # GA sprendimas: ar įjungti kompresiją (mažas latentinis matmuo) ar ne (didesnis latentinis matmuo)
        self.compression_enabled: bool = random.choice([True, False])

        # HNN Architektūros parametrai (GA evoliucionuoja)
        # Latentinis matmuo priklauso nuo to, ar kompresija įjungta
        latent_dim_initial = Config.HNN_LATENT_DIM_COMPRESSED if self.compression_enabled else Config.BINANCE_DATA_POINTS 
        
        self.hnn_architecture: Dict[str, Any] = {
            "latent_dim_flat": latent_dim_initial, 
            "quantization_bits": random.choice(Config.SUPPORTED_QUANTIZATION_BITS),
            "encoder_layers": [
                {"type": "linear", "out_features": random.randint(32, 64), "activation": "GELU"},
                {"type": "linear", "out_features": latent_dim_initial, "activation": "None"}
            ],
            "decoder_layers": [
                {"type": "linear", "out_features": random.randint(32, 64), "activation": "GELU"},
                {"type": "linear", "out_features": Config.BINANCE_DATA_POINTS, "activation": "None"}
            ]
        }
        
        # AI Agentų Sąveikos parametrai (GA evoliucionuoja)
        # Kiekvienas AI agentas turi savo aktyvavimo slenkstį ir įtakos svorį
        self.ai_agent_params: Dict[str, Dict[str, Any]] = {
            agent_name: {
                "active_threshold": random.uniform(0.1, 1.0), # Jei virš šio, agentas bus aktyvuotas
                "influence_weight": random.uniform(0.1, 1.0) # Kaip stipriai jo atsakas veikia fitnesą
            } for agent_name in Config.SUPPORTED_AI_AGENTS
        }
        self.ai_query_strategy: str = random.choice(["sentiment", "architecture_advice", "resource_optimization"])

        # Binance Duomenų Apdorojimo Parametrai (GA evoliucionuoja)
        self.binance_data_params: Dict[str, Any] = {
            "data_points_to_use": Config.BINANCE_DATA_POINTS, # GA gali keisti kiek taškų naudoja
            "normalization_method": random.choice(["min_max", "z_score", "none"]),
            "observe_symbol": random.choice(["BTCUSDT", "ETHUSDT"]) # GA gali pasirinkti simbolį
        }

        # VRAM Valdymo strategijos (GA evoliucionuoja)
        self.vram_optimization_strategy: str = random.choice(["aggressive_quant", "dynamic_resizing", "prioritize_speed"])
        
        # Fitneso Funkcijos svorio parametrai (GA evoliucionuoja SAVO PATIES tikslus!)
        self.fitness_weights: Dict[str, float] = {
            "hnn_accuracy": random.uniform(0.1, 1.0),      # Kuo didesnis tikslumas, tuo geriau
            "vram_efficiency": random.uniform(0.1, 1.0),   # Kuo mažesnis VRAM, tuo geriau
            "compression_ratio_impact": random.uniform(0.1, 1.0), # Kuo didesnė kompresija, tuo geriau (arba tinkamesnis kompresijos pasirinkimas)
            "ai_guidance_bonus": random.uniform(0.1, 1.0)  # Kuo daugiau naudos iš AI, tuo geriau
        }
        # Normalizuojame svorius, kad suma būtų 1.0 (arba tiesiog naudojame kaip koeficientus)
        total_weight = sum(self.fitness_weights.values())
        if total_weight > 0:
            for key in self.fitness_weights:
                self.fitness_weights[key] /= total_weight
        else:
            logging.warning(f"Genomo {self.id} fitneso svorių suma lygi nuliui, gali paveikti fitneso skaičiavimą.")

    def __repr__(self):
        comp_status = "ON" if self.compression_enabled else "OFF"
        return (f"Genome(ID: {self.id[:8]}, Fitness: {self.fitness:.4f}, "
                f"MSE: {self.hnn_mse_loss:.4f}, Comp: {self.compression_ratio:.2f}x, "
                f"VRAM: {self.vram_used_mb:.2f}MB, "
                f"Quant: {self.hnn_architecture['quantization_bits']}bit, "
                f"Comp_Enabled: {comp_status}, Latent: {self.hnn_architecture['latent_dim_flat']}, "
                f"AI Strat: {self.ai_query_strategy})")

# --- HNN statytojas ---
class HNNBuilder:
    @staticmethod
    def build_model_from_genome(genome: Genome, input_dim: int, device: torch.device):
        """
        Sukuriamas PyTorch HNN modelis pagal genomo aprašytą architektūrą.
        Latentinis matmuo priklauso nuo genome.compression_enabled.
        """
        logging.info(f"[Genome {genome.id}] Kuriama HNN architektūra pagal genomą (Compression: {'Enabled' if genome.compression_enabled else 'Disabled'})...")
        
        latent_dim = genome.hnn_architecture["latent_dim_flat"]

        # Encoder sluoksniai
        encoder_layers = []
        current_dim = input_dim
        for i, layer_def in enumerate(genome.hnn_architecture["encoder_layers"]):
            # Jei pasiekiamas latentinis sluoksnis, koreguojame out_features
            if layer_def["out_features"] == latent_dim and i == len(genome.hnn_architecture["encoder_layers"]) - 1:
                 encoder_layers.append(nn.Linear(current_dim, latent_dim))
            else:
                 encoder_layers.append(nn.Linear(current_dim, layer_def["out_features"]))
            
            current_dim = layer_def["out_features"] # Update current_dim for next layer (or latent_dim if it was final)
            if layer_def["activation"] != "None":
                encoder_layers.append(HNNBuilder._get_activation(layer_def["activation"]))
            logging.debug(f"[Genome {genome.id}] Encoder layer {i}: {encoder_layers[-1]}")
        encoder = nn.Sequential(*encoder_layers)

        # Decoder sluoksniai
        decoder_layers = []
        current_dim = latent_dim # Decoder prasideda nuo latentinio matmens
        for i, layer_def in enumerate(genome.hnn_architecture["decoder_layers"]):
            decoder_layers.append(nn.Linear(current_dim, layer_def["out_features"]))
            current_dim = layer_def["out_features"]
            if layer_def["activation"] != "None":
                decoder_layers.append(HNNBuilder._get_activation(layer_def["activation"]))
            logging.debug(f"[Genome {genome.id}] Decoder layer {i}: {decoder_layers[-1]}")
        decoder = nn.Sequential(*decoder_layers)

        model = HNN_Core(encoder, decoder, latent_dim, genome.hnn_architecture["quantization_bits"], genome.id)
        model.to(device) # Perkelia modelį į GPU, jei prieinama
        logging.info(f"[Genome {genome.id}] HNN modelis sukurtas. Latentinis matmuo: {latent_dim}, Kvantavimo bitai: {genome.hnn_architecture['quantization_bits']}")
        return model

    @staticmethod
    def _get_activation(name: str):
        """Grąžina aktyvacijos funkcijos modulį."""
        if name == "GELU":
            return nn.GELU()
        elif name == "ReLU":
            return nn.ReLU()
        elif name == "Sigmoid":
            return nn.Sigmoid()
        elif name == "None":
            return nn.Identity() # Nėra aktyvacijos
        else:
            logging.warning(f"Nepalaikoma aktyvacijos funkcija: {name}. Naudojama Identity.")
            return nn.Identity()

# --- HNN Branduolys (Encoder ir Decoder) ---
class HNN_Core(nn.Module):
    def __init__(self, encoder: nn.Module, decoder: nn.Module, latent_dim: int, quantization_bits: int, genome_id: str):
        super(HNN_Core, self).__init__()
        self.encoder = encoder
        self.decoder = decoder
        self.latent_dim = latent_dim
        self.quantization_bits = quantization_bits
        self.genome_id = genome_id
        
        self._apply_quantization() # Kvantavimo taikymas po modelio sukūrimo
        check_vram_limit("HNN_Core inicializavimas", self.genome_id)

    def _apply_quantization(self):
        """Taikomas dinaminis kvantavimas modeliui, simuliuojant 4-bit."""
        if not torch.cuda.is_available():
            logging.warning(f"[Genome {self.genome_id}] CUDA neprieinama, kvantavimas praleidžiamas arba atliekamas CPU.")
            if self.quantization_bits == 8:
                 try:
                    self.encoder = torch.quantization.quantize_dynamic(self.encoder, {nn.Linear}, dtype=torch.qint8)
                    self.decoder = torch.quantization.quantize_dynamic(self.decoder, {nn.Linear}, dtype=torch.qint8)
                    logging.info(f"[Genome {self.genome_id}] HNN modelis dinamiškai kvantuotas į int8 (CPU).")
                 except Exception as e:
                    logging.error(f"[Genome {self.genome_id}] Nepavyko atlikti dinaminio int8 kvantavimo CPU: {e}. Modelis paliktas float32.")
            elif self.quantization_bits == 4:
                # Simuliuojame 4-bit, modelis lieka float32, bet fitnesas atsižvelgs
                logging.warning(f"[Genome {self.genome_id}] 4 bitų kvantavimas simuliuojamas. Realiai modelis veikia float32. Fitnesas įvertins galimą MSE padidėjimą dėl 'agresyvios' kompresijos.")
            return

        # Jei CUDA prieinama
        if self.quantization_bits == 16:
            self.half() # Konvertuoja visą modelį į float16
            logging.info(f"[Genome {self.genome_id}] HNN modelis konvertuotas į float16 (pusiau tikslumas).")
        elif self.quantization_bits == 8:
            logging.warning(f"[Genome {self.genome_id}] 8 bitų kvantavimas su CUDA paprastai reikalauja QAT. Šiame prototipe tiesiog naudosime dynamic quantize for CPU emulation, arba modelis liks float32, jei nepalaiko GPU. Konvertuojama į float16 kaip kompromisas, jei CPU int8 nepalaikoma.")
            self.half() # Konvertuojame į float16
        elif self.quantization_bits == 4:
            # Simuliuojame 4-bit kvantavimą GPU, modelis lieka float32, bet fitnesas tai įvertina.
            logging.warning(f"[Genome {self.genome_id}] 4 bitų kvantavimas GPU simuliuojamas. Modelis veikia float32. Fitnesas įvertins potencialų tikslumo pablogėjimą, priklausomai nuo GA tikslų.")


    def forward(self, x: torch.Tensor) -> Tuple[torch.Tensor, torch.Tensor]:
        """Praleidžia duomenis per Encoderį ir Decoderį."""
        device = next(self.parameters()).device
        x = x.to(device)

        # Užtikriname tinkamą duomenų tipą pagal kvantavimo nustatymus, kad būtų išvengta klaidų
        if self.quantization_bits == 16:
            x = x.to(torch.float16)
        else:
            x = x.to(torch.float32)

        encoded = self.encoder(x)
        check_vram_limit("HNN Encoderis", self.genome_id)

        decoded = self.decoder(encoded)
        check_vram_limit("HNN Decoderis", self.genome_id)

        return encoded, decoded

# --- Binance Duomenų Tiekėjas (simuliuojamas tiesioginėmis užklausomis) ---
class BinanceStreamManager:
    """
    Simuliuoja Binance duomenų srautą. Vėliau bus pakeista realia TIESIOGINE API integracija.
    GA gali nustatyti, kiek taškų ir kokiu dažniu gauti.
    """
    def __init__(self, genome_id: str):
        self.genome_id = genome_id
        logging.info(f"[Genome {self.genome_id}] BinanceStreamManager inicializuotas (simuliacinis režimas).")

    # Ateityje čia būtų tiesioginės HTTP užklausos, pvz.:
    # async def _fetch_data_from_binance_api(self, symbol: str, limit: int) -> List[float]:
    #     url = f"https://api.binance.com/api/v3/klines?symbol={symbol}&interval=1m&limit={limit}"
    #     async with httpx.AsyncClient() as client: # Reikėtų httpx bibliotekos, kurią GA "leistų"
    #         response = await client.get(url, timeout=5)
    #         response.raise_for_status()
    #         klines = response.json()
    #         # Pavyzdys: grąžina uždarymo kainas
    #         return [float(kline[4]) for kline in klines] 

    def get_simulated_data(self, num_points: int, normalization_method: str) -> torch.Tensor:
        """Grąžina simuliuotus Binance duomenis ir pritaiko normalizaciją."""
        data = torch.randn(1, num_points) # Batch=1

        if normalization_method == "min_max":
            min_val = data.min()
            max_val = data.max()
            if max_val - min_val > 0:
                data = (data - min_val) / (max_val - min_val)
            logging.debug(f"[Genome {self.genome_id}] Duomenys normalizuoti Min-Max metodu.")
        elif normalization_method == "z_score":
            mean = data.mean()
            std = data.std()
            if std > 0:
                data = (data - mean) / std
            logging.debug(f"[Genome {self.genome_id}] Duomenys normalizuoti Z-Score metodu.")
        else: # "none"
            logging.debug(f"[Genome {self.genome_id}] Duomenys nenormalizuoti.")

        logging.debug(f"[Genome {self.genome_id}] Sugeneruoti {num_points} simuliuotų Binance duomenų taškai. Forma: {data.shape}")
        return data

# --- Genetic Architect: Pagrindinis Evoliucijos Variklis ---
class GeneticArchitect:
    def __init__(self):
        self.config = Config()
        self.population: List[Genome] = []
        self.device = torch.device("cuda" if torch.cuda.is_available() else "cpu")
        logging.info(f"GeneticArchitect inicializuotas. Įrenginys: {self.device}")
        
        # Pirminei populiacijai sukurti
        for _ in range(self.config.POPULATION_SIZE):
            self.population.append(Genome())
        logging.info(f"Populiacija inicializuota. Dydis: {len(self.population)}")

    async def evaluate_genome(self, genome: Genome):
        """
        Įvertina vieną genomą, įgyvendindamas visas jo apibrėžtas sistemos dalis.
        VRAM ir kompresija yra tikrinami realiu laiku.
        """
        logging.info(f"\n--- Pradedamas genomo {genome.id[:8]} vertinimas (Compression: {'Enabled' if genome.compression_enabled else 'Disabled'}) ---")
        
        # Užtikriname, kad VRAM būtų išvalyta prieš kiekvieną modelio kūrimą ir treniravimą
        if torch.cuda.is_available():
            torch.cuda.empty_cache()
            logging.info(f"[Genome {genome.id[:8]}] CUDA cache išvalyta prieš HNN kūrimą.")

        try:
            # --- 1. HNN Modelio Kūrimas ir Treniravimas pagal Genomą ---
            hnn_model = HNNBuilder.build_model_from_genome(
                genome=genome,
                input_dim=genome.binance_data_params["data_points_to_use"], 
                device=self.device
            )

            # --- 2. Binance Duomenų Gavimas ir Apdorojimas ---
            binance_manager = BinanceStreamManager(genome.id)
            data = binance_manager.get_simulated_data(
                num_points=genome.binance_data_params["data_points_to_use"],
                normalization_method=genome.binance_data_params["normalization_method"]
            )
            logging.info(f"[Genome {genome.id[:8]}] Duomenų šaltinio simbolis (simuliuotas): {genome.binance_data_params['observe_symbol']}")

            # --- 3. HNN Treniravimas (minimalus, kad matytume MSE pokytį) ---
            optimizer = torch.optim.Adam(hnn_model.parameters(), lr=0.001)
            criterion = nn.MSELoss()

            for epoch in range(self.config.HNN_TRAINING_EPOCHS):
                optimizer.zero_grad()
                encoded, decoded = hnn_model(data)
                loss = criterion(decoded, data.to(self.device)) 
                loss.backward()
                optimizer.step()
                logging.debug(f"[Genome {genome.id[:8]}] HNN treniravimas, Epocha {epoch+1}/{self.config.HNN_TRAINING_EPOCHS}, Loss: {loss.item():.6f}")
                
                if not check_vram_limit(f"HNN Treniravimas Epocha {epoch+1}", genome.id[:8]):
                    genome.hnn_mse_loss = float('inf') 
                    genome.vram_used_mb = Config.VRAM_LIMIT_MB + 1 
                    logging.error(f"[Genome {genome.id[:8]}] HNN treniravimas nutrauktas dėl VRAM viršijimo (genomo VRAM strategija: {genome.vram_optimization_strategy}).")
                    return 

            genome.hnn_mse_loss = loss.item()
            logging.info(f"[Genome {genome.id[:8]}] HNN treniravimas baigtas. Galutinis MSE Loss: {genome.hnn_mse_loss:.6f}")

            # --- 4. Kompresijos Laipsnio Apskaičiavimas ---
            input_size_bytes = genome.binance_data_params["data_points_to_use"] * 4 # Float32
            
            bytes_per_value = genome.hnn_architecture["quantization_bits"] / 8
            latent_size_bytes = genome.hnn_architecture["latent_dim_flat"] * bytes_per_value
            
            if latent_size_bytes > 0:
                genome.compression_ratio = input_size_bytes / latent_size_bytes
            else:
                genome.compression_ratio = 0.0 # Išvengti dalijimo iš nulio

            logging.info(f"[Genome {genome.id[:8]}] Įvesties dydis: {input_size_bytes} baitų. Hologramos dydis: {latent_size_bytes:.2f} baitų ({genome.hnn_architecture['quantization_bits']} bit). Kompresijos laipsnis: {genome.compression_ratio:.2f}x")

            # --- 5. AI Agentų Konsultacijos pagal Genomą ---
            ai_manager = AIAgentsManager(genome.id)
            ai_responses = {}
            total_ai_guidance_score = 0.0

            for agent_name, params in genome.ai_agent_params.items():
                if params["active_threshold"] > random.random(): # Simuliuojamas aktyvavimo slenkstis
                    response = await ai_manager.query_ai_agent(agent_name, genome.ai_query_strategy)
                    ai_responses[agent_name] = response
                    
                    if response['status'] == 'success':
                        if genome.ai_query_strategy == "architecture_advice":
                            total_ai_guidance_score += response['architecture_advice_score'] * params["influence_weight"]
                        elif genome.ai_query_strategy == "sentiment":
                            total_ai_guidance_score += (response['sentiment_score'] + 1.0) / 2.0 * params["influence_weight"] 
                        else: # resource_optimization
                            total_ai_guidance_score += (response['resource_optimization_advice_score'] + 0.5) * params["influence_weight"]
                else:
                    logging.info(f"[Genome {genome.id[:8]}] AI agentas '{agent_name}' neaktyvuotas pagal genomo slenkstį ({params['active_threshold']:.2f}).")
            
            ai_bonus = min(1.0, total_ai_guidance_score) 
            logging.info(f"[Genome {genome.id[:8]}] Visas AI konsultacijos bonusas: {ai_bonus:.4f}")

            # --- 6. Fitneso Apskaičiavimas (su genomo evoliucionuotais svoriais ir kompresijos sprendimu) ---
            genome.vram_used_mb = get_current_vram_usage_mb()

            accuracy_score = max(0, 1.0 - genome.hnn_mse_loss / (1.0 if genome.hnn_mse_loss < 1.0 else genome.hnn_mse_loss)) # Normalizuojame, kad didesnis būtų geriau, virš 1.0 mažėja lėčiau
            vram_score = max(0, 1.0 - genome.vram_used_mb / Config.VRAM_LIMIT_MB) 
            
            # Kompresijos įtakos balo apibrėžimas:
            compression_score = 0.0
            if genome.compression_enabled:
                # Jei kompresija įjungta, skatiname didelę kompresiją, bet ne per daug.
                # Max 800x, tai normalize iki 1.0 su 800x
                compression_score = min(1.0, genome.compression_ratio / 800.0) 
                logging.debug(f"[Genome {genome.id[:8]}] Compression Score (enabled): {compression_score:.4f}")
            else:
                # Jei kompresija išjungta, bet latentinis matmuo per didelis (reiškia prastą "nekompresijos" sprendimą)
                if genome.hnn_architecture["latent_dim_flat"] > Config.BINANCE_DATA_POINTS * 1.2: # Pvz., daugiau nei 20% didesnis latentas
                    compression_score = -0.5 # Bausmė už neefektyvią nekompresiją
                    logging.debug(f"[Genome {genome.id[:8]}] Compression Score (disabled, but inefficient latent): {compression_score:.4f}")
                else:
                    compression_score = 0.5 # Neutralus/nedidelis bonusas už sąmoningą nekompresiją, jei efektyvu
                    logging.debug(f"[Genome {genome.id[:8]}] Compression Score (disabled, efficient latent): {compression_score:.4f}")
            
            genome.fitness = (
                accuracy_score * genome.fitness_weights["hnn_accuracy"] +
                vram_score * genome.fitness_weights["vram_efficiency"] +
                compression_score * genome.fitness_weights["compression_ratio_impact"] +
                ai_bonus * genome.fitness_weights["ai_guidance_bonus"]
            )
            logging.info(f"[Genome {genome.id[:8]}] Fitneso komponentai: Accuracy: {accuracy_score:.4f}, VRAM Score: {vram_score:.4f}, Compression Score: {compression_score:.4f}, AI Bonus: {ai_bonus:.4f}")
            logging.info(f"[Genome {genome.id[:8]}] Apskaičiuotas fitnesas: {genome.fitness:.4f}")

        except MemoryError as e:
            logging.error(f"[Genome {genome.id[:8]}] Kritinė klaida dėl VRAM: {e}. Genomo fitnesas nustatomas į -inf. VRAM strategija: {genome.vram_optimization_strategy}", exc_info=True)
            genome.fitness = -float('inf') 
        except Exception as e:
            logging.error(f"[Genome {genome.id[:8]}] Nepavyko įvertinti genomo dėl netikėtos klaidos: {e}", exc_info=True)
            genome.fitness = -float('inf') 
        finally:
            if 'hnn_model' in locals():
                del hnn_model
            if 'data' in locals():
                del data
            if 'encoded' in locals():
                del encoded
            if 'decoded' in locals():
                del decoded
            if torch.cuda.is_available():
                torch.cuda.empty_cache()
            logging.info(f"--- Genomo {genome.id[:8]} vertinimas baigtas. VRAM po valymo: {get_current_vram_usage_mb():.2f} MB ---")

    async def evolve(self):
        """Pagrindinė GA evoliucijos kilpa."""
        logging.info("Pradedama GeneticArchitect evoliucija...")

        for generation in range(self.config.NUM_GENERATIONS):
            logging.info(f"\n===== Generacija {generation + 1}/{self.config.NUM_GENERATIONS} =====")
            
            tasks = [self.evaluate_genome(genome) for genome in self.population]
            await asyncio.gather(*tasks) 

            self.population.sort(key=lambda g: g.fitness, reverse=True)
            logging.info(f"Generacijos {generation + 1} geriausias fitnesas: {self.population[0].fitness:.4f}")
            logging.info(f"Generacijos {generation + 1} blogiausias fitnesas: {self.population[-1].fitness:.4f}")
            logging.info(f"Generacijos {generation + 1} geriausias genomas: {self.population[0]}")

            new_population = []
            new_population.append(self.population[0]) # Elitizmas
            
            while len(new_population) < self.config.POPULATION_SIZE:
                parent1 = random.choice(self.population[:Config.POPULATION_SIZE // 2]) 
                parent2 = random.choice(self.population[:Config.POPULATION_SIZE // 2])

                child1, child2 = self.crossover(parent1, parent2)
                
                self.mutate(child1)
                self.mutate(child2)

                new_population.append(child1)
                if len(new_population) < self.config.POPULATION_SIZE:
                    new_population.append(child2)
            
            self.population = new_population
            logging.info(f"Generacija {generation + 1} baigta. Nauja populiacija sukurta.")

        logging.info("\n=== GeneticArchitect evoliucija baigta ===")
        logging.info(f"Galutinis geriausias genomas: {self.population[0]}")

    def crossover(self, parent1: Genome, parent2: Genome) -> Tuple[Genome, Genome]:
        """Atlieka kryžminimą tarp dviejų genomų."""
        child1 = Genome(genome_id=str(uuid.uuid4()))
        child2 = Genome(genome_id=str(uuid.uuid4()))

        # Atstatome latent_dim_flat, kad būtų dinamiškai nustatytas pagal compression_enabled
        child1.hnn_architecture["latent_dim_flat"] = Config.HNN_LATENT_DIM_COMPRESSED if child1.compression_enabled else Config.BINANCE_DATA_POINTS
        child2.hnn_architecture["latent_dim_flat"] = Config.HNN_LATENT_DIM_COMPRESSED if child2.compression_enabled else Config.BINANCE_DATA_POINTS


        if random.random() < Config.CROSSOVER_RATE:
            # Kompresijos įjungimas/išjungimas
            child1.compression_enabled = random.choice([parent1.compression_enabled, parent2.compression_enabled])
            child2.compression_enabled = random.choice([parent1.compression_enabled, parent2.compression_enabled])
            
            # Latent dim (priklausys nuo compression_enabled)
            child1.hnn_architecture["latent_dim_flat"] = Config.HNN_LATENT_DIM_COMPRESSED if child1.compression_enabled else Config.BINANCE_DATA_POINTS
            child2.hnn_architecture["latent_dim_flat"] = Config.HNN_LATENT_DIM_COMPRESSED if child2.compression_enabled else Config.BINANCE_DATA_POINTS

            # Kvantavimo bitai
            child1.hnn_architecture["quantization_bits"] = random.choice([parent1.hnn_architecture["quantization_bits"], parent2.hnn_architecture["quantization_bits"]])
            child2.hnn_architecture["quantization_bits"] = random.choice([parent1.hnn_architecture["quantization_bits"], parent2.hnn_architecture["quantization_bits"]])
            
            # Encoder/Decoder sluoksniai (paprastas swap'as dėl sudėtingumo)
            if random.random() < 0.5:
                child1.hnn_architecture["encoder_layers"], child2.hnn_architecture["encoder_layers"] = parent1.hnn_architecture["encoder_layers"], parent2.hnn_architecture["encoder_layers"]
            else:
                child1.hnn_architecture["encoder_layers"], child2.hnn_architecture["encoder_layers"] = parent2.hnn_architecture["encoder_layers"], parent1.hnn_architecture["encoder_layers"]

            if random.random() < 0.5:
                child1.hnn_architecture["decoder_layers"], child2.hnn_architecture["decoder_layers"] = parent1.hnn_architecture["decoder_layers"], parent2.hnn_architecture["decoder_layers"]
            else:
                child1.hnn_architecture["decoder_layers"], child2.hnn_architecture["decoder_layers"] = parent2.hnn_architecture["decoder_layers"], parent1.hnn_architecture["decoder_layers"]


        # Kryžminimas AI agentų parametrams
        for agent_name in Config.SUPPORTED_AI_AGENTS:
            for param_key in parent1.ai_agent_params[agent_name]:
                child1.ai_agent_params[agent_name][param_key] = random.choice([parent1.ai_agent_params[agent_name][param_key], parent2.ai_agent_params[agent_name][param_key]])
                child2.ai_agent_params[agent_name][param_key] = random.choice([parent1.ai_agent_params[agent_name][param_key], parent2.ai_agent_params[agent_name][param_key]])
        child1.ai_query_strategy = random.choice([parent1.ai_query_strategy, parent2.ai_query_strategy])
        child2.ai_query_strategy = random.choice([parent1.ai_query_strategy, parent2.ai_query_strategy])

        # Kryžminimas Binance duomenų parametrams
        for key in parent1.binance_data_params:
            child1.binance_data_params[key] = random.choice([parent1.binance_data_params[key], parent2.binance_data_params[key]])
            child2.binance_data_params[key] = random.choice([parent1.binance_data_params[key], parent2.binance_data_params[key]])
        
        # Kryžminimas fitneso svoriams
        for key in parent1.fitness_weights:
            child1.fitness_weights[key] = random.choice([parent1.fitness_weights[key], parent2.fitness_weights[key]])
            child2.fitness_weights[key] = random.choice([parent1.fitness_weights[key], parent2.fitness_weights[key]])
        
        logging.debug(f"Kryžminimas tarp {parent1.id[:8]} ir {parent2.id[:8]} sukūrė {child1.id[:8]} ir {child2.id[:8]}")
        return child1, child2

    def mutate(self, genome: Genome):
        """Atlieka mutaciją genomui."""
        if random.random() < Config.MUTATION_RATE:
            logging.debug(f"Mutacija vykdoma genomui {genome.id[:8]}")
            
            # Mutuojame kompresijos įjungimą/išjungimą (GA sprendimas)
            if random.random() < Config.MUTATION_RATE_COMPRESSION_TOGGLE:
                genome.compression_enabled = not genome.compression_enabled
                if not genome.compression_enabled:
                    # Jei kompresija išjungta, latentinis matmuo tampa artimas įvesties matmeniui
                    genome.hnn_architecture["latent_dim_flat"] = random.randint(Config.BINANCE_DATA_POINTS - 10, Config.BINANCE_DATA_POINTS + 10) # Leidžiame šiek tiek variacijos
                else:
                    # Jei kompresija įjungta, latentinis matmuo grįžta į mažą vertę
                    genome.hnn_architecture["latent_dim_flat"] = Config.HNN_LATENT_DIM_COMPRESSED
                logging.debug(f"  Mutavo compression_enabled į {genome.compression_enabled}, latent_dim_flat į {genome.hnn_architecture['latent_dim_flat']}")

            # Mutuojame HNN latentinę dimensiją (tik jei kompresija įjungta)
            # Dabar latent_dim_flat yra tiesiogiai susietas su compression_enabled, todėl šis blokas mažiau reikalingas,
            # bet palieku dėl galimybės vėliau leisti GA labiau kontroliuoti latentinį dydį.
            if genome.compression_enabled and random.random() < 0.1:
                genome.hnn_architecture["latent_dim_flat"] = random.choice([Config.HNN_LATENT_DIM_COMPRESSED, Config.HNN_LATENT_DIM_COMPRESSED + 1]) # Leidžiame nedidelius pokyčius
                logging.debug(f"  Mutavo latent_dim_flat (kai compression_enabled) į {genome.hnn_architecture['latent_dim_flat']}")


            # Mutuojame kvantavimo bitus
            if random.random() < 0.2:
                genome.hnn_architecture["quantization_bits"] = random.choice(Config.SUPPORTED_QUANTIZATION_BITS)
                logging.debug(f"  Mutavo quantization_bits į {genome.hnn_architecture['quantization_bits']}")

            # Mutuojame encoder/decoder sluoksnių out_features
            if random.random() < 0.1:
                if genome.hnn_architecture["encoder_layers"]:
                    # Sluoksnių matmenys negali būti mažesni už latent_dim, jei tai nėra pats latent_dim
                    current_out = genome.hnn_architecture["encoder_layers"][0]["out_features"]
                    new_out = random.randint(max(32, genome.hnn_architecture["latent_dim_flat"]), 128)
                    genome.hnn_architecture["encoder_layers"][0]["out_features"] = new_out
                    logging.debug(f"  Mutavo encoder_layers[0]['out_features'] į {genome.hnn_architecture['encoder_layers'][0]['out_features']}")
            if random.random() < 0.1:
                if genome.hnn_architecture["decoder_layers"]:
                    current_out = genome.hnn_architecture["decoder_layers"][0]["out_features"]
                    new_out = random.randint(max(32, genome.hnn_architecture["latent_dim_flat"]), 128)
                    genome.hnn_architecture["decoder_layers"][0]["out_features"] = new_out
                    logging.debug(f"  Mutavo decoder_layers[0]['out_features'] į {genome.hnn_architecture['decoder_layers'][0]['out_features']}")
            
            # Mutuojame AI agentų parametrus
            if random.random() < 0.2:
                agent_to_mutate = random.choice(Config.SUPPORTED_AI_AGENTS)
                param_to_mutate = random.choice(list(genome.ai_agent_params[agent_to_mutate].keys()))
                genome.ai_agent_params[agent_to_mutate][param_to_mutate] = random.uniform(0.1, 1.0)
                logging.debug(f"  Mutavo AI agento '{agent_to_mutate}' parametrą '{param_to_mutate}' į {genome.ai_agent_params[agent_to_mutate][param_to_mutate]:.2f}")
            if random.random() < 0.1: # Mutuojame AI užklausos strategiją
                genome.ai_query_strategy = random.choice(["sentiment", "architecture_advice", "resource_optimization"])
                logging.debug(f"  Mutavo AI užklausos strategiją į {genome.ai_query_strategy}")

            # Mutuojame Binance duomenų parametrus
            if random.random() < 0.1:
                key_to_mutate = random.choice(list(genome.binance_data_params.keys()))
                if key_to_mutate == "data_points_to_use":
                    genome.binance_data_params[key_to_mutate] = random.randint(50, 200) 
                elif key_to_mutate == "normalization_method":
                    genome.binance_data_params[key_to_mutate] = random.choice(["min_max", "z_score", "none"])
                elif key_to_mutate == "observe_symbol":
                    genome.binance_data_params[key_to_mutate] = random.choice(["BTCUSDT", "ETHUSDT", "SOLUSDT"]) 
                logging.debug(f"  Mutavo Binance duomenų parametrą '{key_to_mutate}' į {genome.binance_data_params[key_to_mutate]}")

            # Mutuojame VRAM optimizavimo strategiją
            if random.random() < 0.1:
                genome.vram_optimization_strategy = random.choice(["aggressive_quant", "dynamic_resizing", "prioritize_speed"])
                logging.debug(f"  Mutavo VRAM optimizavimo strategiją į {genome.vram_optimization_strategy}")

            # Mutuojame fitneso svorius (GA mokosi pats keisti savo tikslus)
            if random.random() < 0.2:
                key_to_mutate = random.choice(list(genome.fitness_weights.keys()))
                genome.fitness_weights[key_to_mutate] = random.uniform(0.1, 1.0)
                total_weight = sum(genome.fitness_weights.values())
                if total_weight > 0:
                    for key in genome.fitness_weights:
                        genome.fitness_weights[key] /= total_weight
                logging.debug(f"  Mutavo fitneso svorį '{key_to_mutate}' į {genome.fitness_weights[key_to_mutate]:.2f}")


# --- Pagrindinė Vykdymo Funkcija ---
async def main():
    ga = GeneticArchitect()
    await ga.evolve()

if __name__ == "__main__":
    logging.info("--- Pradedamas GeneticArchitect sistemos vykdymas ---")
    asyncio.run(main())
    logging.info("--- GeneticArchitect sistemos vykdymas baigtas ---")

   - **No CUDA:** VRAM=0, vram_score=1.0 – netestuoja. Realybėje su ON (mažas modelis) – mažas VRAM, OFF – didesnis.
