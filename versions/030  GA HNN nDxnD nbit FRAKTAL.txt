class SystemConfig:
    device: str = "cuda" if torch.cuda.is_available() else "cpu"
    initial_input_shape: Tuple[int, ...] = (1, 8, 8)
    supported_qat_bits: List[int] = [8]

# --- Genomo klasė, apibrėžianti tinklo architektūrą ---
class LayerDefinition:
    def __init__(self, type: str = "conv", activation: str = "ReLU", out_features: int = 64, dropout_rate: float = 0.0,
                 kernel_size: int = 3, stride: int = 1):
        self.type = type
        self.activation = activation
        self.out_features = out_features
        self.dropout_rate = dropout_rate
        self.kernel_size = kernel_size
        self.stride = stride
    
    def to_dict(self):
        return self.__dict__
    
    @staticmethod
    def from_dict(d: Dict):
        return LayerDefinition(**d)

class Genome:
    def __init__(self):
        self.id: str = str(random.getrandbits(64))
        self.hnn_architecture: Dict[str, Union[int, List[LayerDefinition]]] = {
            "latent_dim_flat": random.choice([64, 128, 256]),
            "encoder_layers": [LayerDefinition() for _ in range(random.randint(1, 3))],

        }
        
        self.layer_bits: Dict[str, List[int]] = {

            "decoder": [random.randint(1, 512) for _ in self.hnn_architecture["decoder_layers"]]
        }
        
        self.learning_rate: float = random.uniform(1e-4, 1e-3)
        self.batch_size: int = random.choice([8, 16])
        self.training_epochs: int = random.randint(1, 2)
        self.qat_enabled: bool = random.choice([True, False])

        self.loss: float = float('inf')
        self.accuracy: float = 0.0
        self.fitness: float = -float('inf')
        self.compression_ratio: float = 0.0
        
        self.vram_used_mb: float = 0.0
        self.vram_limit_mb: int = 2048
        self.vram_warning_threshold: float = 0.8
        self.min_time_between_vram_clears_sec: int = 60
        
        self.fitness_weights = {
            "hnn_accuracy": random.uniform(0.1, 1.0),
            "vram_efficiency": random.uniform(0.1, 1.0),
            "compression_ratio_impact": random.uniform(0.1, 1.0),
            "ai_guidance_bonus": random.uniform(0.1, 1.0)
        }
        self._normalize_fitness_weights()

        # Nauji evoliucionuojami koeficientai
        self.mutation_threshold: float = 0.1  # Std dev riba mutation_rate adaptacijai
        self.niche_radius: float = 0.2  # Panašumo riba niche penalty

        self.params /= np.linalg.norm(self.params) if np.linalg.norm(self.params) != 0 else 1  # Unit sphere

    def _normalize_fitness_weights(self):
        total_weight = sum(self.fitness_weights.values())
        if total_weight > 0:
            for key in self.fitness_weights:
                self.fitness_weights[key] /= total_weight
        else:
            logger.warning("Fitneso svorių suma lygi nuliui", genome_id=self.id)

# --- HNN Modelis ---
class HNN_Core(nn.Module):
    def __init__(self, encoder: nn.Module, decoder: nn.Module, latent_dim: int, genome: Genome, pre_flatten_shapes: List[Tuple[int, ...]]):
        super().__init__()
        self.encoder = encoder
        self.decoder = decoder
        self.latent_dim = latent_dim
        self.genome = genome
        self.pre_flatten_shapes = pre_flatten_shapes
        self.device = torch.device(SystemConfig.device)
        self.supported_qat_bits = SystemConfig.supported_qat_bits
        self.initial_input_shape = SystemConfig.initial_input_shape

    def custom_quantize(self, tensor: torch.Tensor, bits: int) -> torch.Tensor:
        if bits >= 32:
            return tensor
        
        max_val = tensor.abs().max()
        if max_val == 0:
            return tensor
            
        scale = (2.0**bits - 1.0) / (2.0 * max_val)
        zero_point = 2.0**(bits-1)
        
        quantized_tensor = torch.round(tensor * scale + zero_point)
        dequantized_tensor = (quantized_tensor - zero_point) / scale
        
        return dequantized_tensor

    def forward(self, x: torch.Tensor) -> Tuple[torch.Tensor, torch.Tensor]:
        if len(x.shape) == len(self.initial_input_shape):
            x = x.unsqueeze(0)
        
        encoded = x
        for i, layer_module in enumerate(self.encoder):
            if not self.genome.qat_enabled and i < len(self.genome.layer_bits.get("encoder", [])):
                bits = self.genome.layer_bits["encoder"][i]
                encoded = self.custom_quantize(encoded, bits)
            encoded = layer_module(encoded)

        decoded = encoded
        for i, layer_module in enumerate(self.decoder):
            if not self.genome.qat_enabled and i < len(self.genome.layer_bits.get("decoder", [])):
                bits = self.genome.layer_bits["decoder"][i]
                decoded = self.custom_quantize(decoded, bits)
            decoded = layer_module(decoded)

        return encoded, decoded

# --- HNNBuilder, skirta kurti modelį iš genomo ---
class HNNBuilder:
    @staticmethod
    def _get_activation(name: str) -> nn.Module:
        activations = {
            "ReLU": nn.ReLU, "GELU": nn.GELU, "Sigmoid": nn.Sigmoid, 
            "LeakyReLU": nn.LeakyReLU, "Tanh": nn.Tanh, "None": nn.Identity
        }
        return activations.get(name, nn.Identity)()

    @staticmethod
    def _calculate_output_dim(input_dims: Union[int, Tuple[int, ...]], kernel_sizes: Union[int, Tuple[int, ...]], strides: Union[int, Tuple[int, ...]], paddings: Union[int, Tuple[int, ...]], transposed: bool = False, output_paddings: Union[int, Tuple[int, ...]] = 0) -> Tuple[int, ...]:
        if isinstance(input_dims, int):
            input_dims = (input_dims,)
            kernel_sizes = (kernel_sizes,)
            strides = (strides,)
            paddings = (paddings,)
            if isinstance(output_paddings, int):
                output_paddings = (output_paddings,)
        return tuple( (d - 1) * s - 2 * p + (k - 1) + op + 1 if transposed else math.floor((d + 2 * p - k) / s) + 1 for d, k, s, p, op in zip(input_dims, kernel_sizes, strides, paddings, output_paddings) )

    @staticmethod
    def build_model_from_genome(genome: Genome, initial_input_shape: Tuple[int, ...], device: torch.device) -> Union[HNN_Core, None]:
        try:
            N = SystemConfig.get_N_dims()
            
            encoder_modules = []
            current_shape = initial_input_shape
            spatial_shapes_stack = []

            for i, layer_def in enumerate(genome.hnn_architecture["encoder_layers"]):
                in_channels = current_shape[0]
                
                if layer_def.type == "conv":
                    conv_class = getattr(nn, f'Conv{N}d')
                    kernel_tuple = (layer_def.kernel_size,) * N
                    stride_tuple = (layer_def.stride,) * N
                    pad_tuple = (layer_def.kernel_size // 2,) * N
                    new_dims = HNNBuilder._calculate_output_dim(current_shape[1:], kernel_tuple, stride_tuple, pad_tuple)
                    encoder_modules.append(conv_class(in_channels, layer_def.out_features, kernel_tuple, stride_tuple, pad_tuple))
                    current_shape = (layer_def.out_features,) + new_dims
                elif layer_def.type == "batch_norm":
                    bn_class = getattr(nn, f'BatchNorm{N}d')
                    encoder_modules.append(bn_class(in_channels))
                elif layer_def.type == "max_pool":
                    pool_class = getattr(nn, f'MaxPool{N}d')
                    kernel_tuple = (layer_def.kernel_size,) * N
                    stride_tuple = (layer_def.stride,) * N
                    pad_tuple = (layer_def.kernel_size // 2,) * N
                    new_dims = HNNBuilder._calculate_output_dim(current_shape[1:], kernel_tuple, stride_tuple, pad_tuple)
                    encoder_modules.append(pool_class(kernel_size=kernel_tuple, stride=stride_tuple, padding=pad_tuple))
                    current_shape = (in_channels,) + new_dims
                elif layer_def.type == "avg_pool":
                    pool_class = getattr(nn, f'AvgPool{N}d')
                    kernel_tuple = (layer_def.kernel_size,) * N
                    stride_tuple = (layer_def.stride,) * N
                    pad_tuple = (layer_def.kernel_size // 2,) * N
                    new_dims = HNNBuilder._calculate_output_dim(current_shape[1:], kernel_tuple, stride_tuple, pad_tuple)
                    encoder_modules.append(pool_class(kernel_size=kernel_tuple, stride=stride_tuple, padding=pad_tuple))
                    current_shape = (in_channels,) + new_dims
                elif layer_def.type == "dropout":
                    encoder_modules.append(nn.Dropout(p=layer_def.dropout_rate))
                elif layer_def.type == "linear":
                    spatial_shapes_stack.append(current_shape)
                    encoder_modules.append(nn.Flatten(start_dim=1))
                    encoder_modules.append(nn.Linear(math.prod(current_shape), layer_def.out_features))
                    current_shape = (layer_def.out_features,)
                
                encoder_modules.append(HNNBuilder._get_activation(layer_def.activation))

            if N > 0 and not isinstance(encoder_modules[-1], nn.Linear):
                spatial_shapes_stack.append(current_shape)
                encoder_modules.append(nn.Flatten(start_dim=1))
                flat_input_dim = math.prod(current_shape)
                encoder_modules.append(nn.Linear(flat_input_dim, genome.hnn_architecture["latent_dim_flat"]))
                current_shape = (genome.hnn_architecture["latent_dim_flat"],)
            else:
                flat_input_dim = current_shape[0]
                if flat_input_dim != genome.hnn_architecture["latent_dim_flat"]:
                    encoder_modules.append(nn.Linear(flat_input_dim, genome.hnn_architecture["latent_dim_flat"]))

            encoder = nn.Sequential(*encoder_modules)

            decoder_modules = []
            
            if not spatial_shapes_stack:
                logger.error("Spatial shapes stack is empty, cannot build decoder.")
                return None
            
            target_shape = spatial_shapes_stack[-1]
            decoder_modules.append(nn.Linear(genome.hnn_architecture["latent_dim_flat"], math.prod(target_shape)))
            spatial_shapes_stack.pop()
            decoder_modules.append(nn.Unflatten(1, target_shape))
            current_shape = target_shape
            N = len(current_shape) - 1

            for i, layer_def in enumerate(genome.hnn_architecture["decoder_layers"]):
                in_channels = current_shape[0]

                if layer_def.type == "conv_transpose":
                    conv_class = getattr(nn, f'ConvTranspose{N}d')
                    kernel_tuple = (layer_def.kernel_size,) * N
                    stride_tuple = (layer_def.stride,) * N
                    pad_tuple = (layer_def.kernel_size // 2,) * N
                    calc_dims = HNNBuilder._calculate_output_dim(current_shape[1:], kernel_tuple, stride_tuple, pad_tuple, transposed=True, output_paddings=0)
                    output_padding_tuple = tuple(max(0, c - d) for c, d in zip(calc_dims, current_shape[1:]))
                    new_dims = HNNBuilder._calculate_output_dim(current_shape[1:], kernel_tuple, stride_tuple, pad_tuple, transposed=True, output_paddings=output_padding_tuple)
                    decoder_modules.append(conv_class(in_channels, layer_def.out_features, kernel_tuple, stride_tuple, pad_tuple, output_padding_tuple))
                    current_shape = (layer_def.out_features,) + new_dims
                elif layer_def.type == "batch_norm":
                    bn_class = getattr(nn, f'BatchNorm{N}d')
                    decoder_modules.append(bn_class(in_channels))
                elif layer_def.type == "upsample":
                    decoder_modules.append(nn.Upsample(scale_factor=layer_def.stride, mode='nearest'))
                    current_shape = (in_channels, *(dim * layer_def.stride for dim in current_shape[1:]))
                elif layer_def.type == "linear":
                    if spatial_shapes_stack:
                        unflatten_shape = spatial_shapes_stack.pop()
                        decoder_modules.append(nn.Unflatten(1, unflatten_shape))
                        current_shape = unflatten_shape
                        N = len(current_shape) - 1
                    decoder_modules.append(nn.Linear(math.prod(current_shape), layer_def.out_features))
                    current_shape = (layer_def.out_features,)
                
                decoder_modules.append(HNNBuilder._get_activation(layer_def.activation))

            if current_shape[0] != initial_input_shape[0]:
                conv_class = getattr(nn, f'Conv{N}d')
                decoder_modules.append(conv_class(current_shape[0], initial_input_shape[0], (1,) * N))
                current_shape = (initial_input_shape[0], *current_shape[1:])

            if math.prod(current_shape[1:]) != math.prod(initial_input_shape[1:]):
                decoder_modules.append(nn.Flatten(start_dim=1))
                decoder_modules.append(nn.Linear(math.prod(current_shape), math.prod(initial_input_shape)))
                decoder_modules.append(nn.Unflatten(1, initial_input_shape))
            
            decoder = nn.Sequential(*decoder_modules)
            
            if not encoder_modules or not decoder_modules:
                logger.error("Nepavyko sukurti modulių.")
                return None

            model = HNN_Core(encoder, decoder, genome.hnn_architecture["latent_dim_flat"], genome, spatial_shapes_stack[::-1])
            return model.to(device)
            
        except Exception as e:
            logger.error(f"Nepavyko sukurti modelio iš genomo: {e}\n{traceback.format_exc()}")
            return None

# --- Genetinis Algoritmas ---
class GeneticArchitect:
    def __init__(self, config: SystemConfig, initial_input_shape: Tuple[int, ...]):
        self.config = config
        self.population: List[Genome] = []
        self.initial_input_shape = initial_input_shape
        self.last_vram_clear_time = time.time()
        self.mutation_rate = 0.1  # Pradinis
        self.crossover_rate = 0.7  # Pradinis
        self.best_fitness_history = []  # Sekti stagnaciją
        self.history = []  # Surrogate istorija: [(params_vec, fitness), ...]
        self.surrogate = None

    async def initialize_population(self, population_size: int):
        self.population = []
        avg_clusters = 5  # Pradinis, vėliau iš genomų
        random_params = [np.random.rand(10) for _ in range(100)]  # 10 params, 100 kandidatų
        kmeans = KMeans(n_clusters=avg_clusters)
        kmeans.fit(random_params)
        labels = kmeans.labels_
        for cluster in range(avg_clusters):
            cluster_params = [p for p, l in zip(random_params, labels) if l == cluster]
            for _ in range(population_size // avg_clusters):
                genome = Genome()
                genome.params = random.choice(cluster_params) if cluster_params else np.random.rand(10)
                attempts = 0
                while attempts < 10:
                    model = HNNBuilder.build_model_from_genome(genome, self.initial_input_shape, torch.device(self.config.device))
                    if model is not None:
                        self.population.append(genome)
                        del model
                        if self.config.device == "cuda":
                            torch.cuda.empty_cache()
                        break
                    genome = Genome()
                    attempts += 1
                if attempts == 10:
                    break
        if not self.population:
            raise RuntimeError("Nepavyko inicializuoti populiacijos.")

        logger.info(f"Populiacija inicializuota su {len(self.population)} genomų.")
    
    def _check_vram_limit(self, genome: Genome):
        if not torch.cuda.is_available():
            return
        torch.cuda.empty_cache()
        allocated_vram = torch.cuda.memory_allocated(0) / (1024 ** 2)
        reserved_vram = torch.cuda.memory_reserved(0) / (1024 ** 2)
        genome.vram_used_mb = allocated_vram
        if reserved_vram > genome.vram_limit_mb * genome.vram_warning_threshold:
            current_time = time.time()
            if (current_time - self.last_vram_clear_time) > genome.min_time_between_vram_clears_sec:
                logger.warning(f"VRAM slenkstis viršytas. Valoma. Rezervuota: {reserved_vram:.2f}MB")
                torch.cuda.empty_cache()
                self.last_vram_clear_time = current_time

    async def evaluate_genome(self, genome: Genome) -> None:
        avg_usage = np.mean([g.surrogate_usage for g in self.population])
        params_vec = np.array([
            genome.hnn_architecture['latent_dim_flat'],
            genome.layer_bits['encoder'][0] if genome.layer_bits['encoder'] else 0,
            genome.layer_bits['decoder'][0] if genome.layer_bits['decoder'] else 0
        ])
        if self.surrogate and random.random() < avg_usage:
            predicted_fitness = self.surrogate.predict([params_vec])[0]
            genome.fitness = predicted_fitness
            logger.info(f"Surrogate prognozė genome {genome.id[:8]}: {predicted_fitness:.4f}")
            return

        device = torch.device(self.config.device)
        dummy_data = torch.randn(genome.batch_size, *self.initial_input_shape).to(device)

        hnn_model = HNNBuilder.build_model_from_genome(genome, self.initial_input_shape, device)
        if hnn_model is None:
            genome.fitness = -float('inf')
            return

        if genome.qat_enabled and 'cpu' not in self.config.device:
            try:
                hnn_model.qconfig = torch.quantization.get_default_qat_qconfig('fbgemm')
                torch.quantization.prepare_qat(hnn_model, inplace=True)
            except Exception as e:
                logger.warning(f"QAT nepavyko: {e}. Išjungiamas.")
                genome.qat_enabled = False

        criterion = nn.MSELoss()
        optimizer = optim.Adam(hnn_model.parameters(), lr=genome.learning_rate)

        hnn_model.train()
        for epoch in range(genome.training_epochs):
            optimizer.zero_grad()
            encoded_output, decoded_output = hnn_model(dummy_data)
            loss = criterion(decoded_output, dummy_data)
            loss.backward()
            optimizer.step()
            self._check_vram_limit(genome)

        if genome.qat_enabled and 'cpu' not in self.config.device:
            try:
                torch.quantization.convert(hnn_model, inplace=True)
            except Exception as e:
                logger.warning(f"QAT konversija nepavyko: {e}.")
                genome.qat_enabled = False
        
        hnn_model.eval()
        with torch.no_grad():
            encoded_val, decoded_val = hnn_model(dummy_data)
            val_loss = criterion(decoded_val, dummy_data)
            accuracy_score = 1.0 - (val_loss.item() / (dummy_data.max() - dummy_data.min()).item() + 1e-8)
            genome.accuracy = max(0.0, min(1.0, accuracy_score))

        genome.loss = val_loss.item()
        
        del hnn_model
        if device.type == "cuda":
            torch.cuda.empty_cache()

        total_bits = sum(genome.layer_bits.get("encoder", []) + genome.layer_bits.get("decoder", []))
        num_layers = len(genome.layer_bits.get("encoder", [])) + len(genome.layer_bits.get("decoder", []))
        avg_bits_per_layer = total_bits / (num_layers + 1e-5)

        input_size_bits = math.prod(self.initial_input_shape) * 32
        latent_size_bits = genome.hnn_architecture["latent_dim_flat"] * avg_bits_per_layer
        genome.compression_ratio = input_size_bits / latent_size_bits if latent_size_bits > 0 else 0.0

        vram_score = max(0.0, min(1.0, 1.0 - genome.vram_used_mb / (genome.vram_limit_mb + 1e-8)))
        if avg_bits_per_layer > genome.large_bit_penalty_threshold:
            vram_score *= genome.large_bit_penalty_factor

        ai_bonus = random.uniform(0.0, 0.1)
        total_weight_sum = sum(genome.fitness_weights.values()) + 1e-8
        normalized_weights = {k: v / total_weight_sum for k, v in genome.fitness_weights.items()}

        if genome.fitness_formula_type == "linear":
            genome.fitness = (genome.accuracy * normalized_weights["hnn_accuracy"] +
                              vram_score * normalized_weights["vram_efficiency"] +
                              genome.compression_ratio * normalized_weights["compression_ratio_impact"] +
                              ai_bonus * normalized_weights["ai_guidance_bonus"])
        # ... (likusios fitneso formulės)
        
        logger.info(f"Genomas {genome.id[:8]}: Fitnesas={genome.fitness:.4f}, Loss={genome.loss:.4f}, Accuracy={genome.accuracy:.4f}, VRAM={genome.vram_used_mb:.2f}MB, Comp.Ratio={genome.compression_ratio:.2f}")

        # Pridėti į istoriją surrogate
        self.history.append((params_vec, genome.fitness))

    def train_surrogate(self):
        if len(self.history) > 20:
            X, y = zip(*self.history)
            X_train, _, y_train, _ = train_test_split(X, y, test_size=0.2)
            self.surrogate = RandomForestRegressor(n_estimators=50)
            self.surrogate.fit(X_train, y_train)
            logger.info("Surrogate apmokytas")

    def apply_niche_penalty(self):
        avg_radius = np.mean([g.niche_radius for g in self.population])
        for i, genome in enumerate(self.population):
            similar_count = 0
            genome_vec = np.array([
                genome.hnn_architecture['latent_dim_flat'],
                genome.layer_bits['encoder'][0] if genome.layer_bits['encoder'] else 0,
                genome.layer_bits['decoder'][0] if genome.layer_bits['decoder'] else 0
            ])
            for other in self.population:
                if genome != other:
                    other_vec = np.array([
                        other.hnn_architecture['latent_dim_flat'],
                        other.layer_bits['encoder'][0] if other.layer_bits['encoder'] else 0,
                        other.layer_bits['decoder'][0] if other.layer_bits['decoder'] else 0
                    ])
                    dist = euclidean(genome_vec, other_vec)
                    if dist < avg_radius:
                        similar_count += 1
            if similar_count > 0:
                genome.fitness /= (1 + similar_count)
                logger.debug(f"Niche penalty genome {genome.id[:8]}: fitness sumažinta {genome.fitness:.4f} dėl {similar_count} panašių")

    def select_parents(self, num_parents: int) -> List[Genome]:
        total_fitness = sum(max(0, g.fitness) for g in self.population)
        if total_fitness == 0:
            return random.sample(self.population, k=num_parents)

        selection_probabilities = [max(0, g.fitness) / total_fitness for g in self.population]
        parents = random.choices(self.population, weights=selection_probabilities, k=num_parents)
        return parents

    def crossover(self, parent1: Genome, parent2: Genome) -> Genome:
        child_genome = Genome()
        for section in ["encoder_layers", "decoder_layers"]:
            min_len = min(len(parent1.hnn_architecture[section]), len(parent2.hnn_architecture[section]))
            if min_len > 0:
                crossover_point = random.randint(1, min_len)
                child_genome.hnn_architecture[section] = parent1.hnn_architecture[section][:crossover_point] + parent2.hnn_architecture[section][crossover_point:]
            else:
                child_genome.hnn_architecture[section] = []

        child_genome.learning_rate = random.choice([parent1.learning_rate, parent2.learning_rate])
        child_genome.batch_size = random.choice([parent1.batch_size, parent2.batch_size])
        child_genome.qat_enabled = random.choice([parent1.qat_enabled, parent2.qat_enabled])
        child_genome.hnn_architecture["latent_dim_flat"] = random.choice([parent1.hnn_architecture["latent_dim_flat"], parent2.hnn_architecture["latent_dim_flat"]])
        child_genome.fitness_formula_type = random.choice([parent1.fitness_formula_type, parent2.fitness_formula_type])
        
        for section in ["encoder", "decoder"]:
            min_len = min(len(parent1.layer_bits.get(section, [])), len(parent2.layer_bits.get(section, [])))
            child_genome.layer_bits[section] = parent1.layer_bits.get(section, [])[:min_len] + parent2.layer_bits.get(section, [])[min_len:]
            
        return child_genome

    def mutate(self, genome: Genome) -> None:
        mutation_strength = random.choice([1, 2, 3])
        for _ in range(mutation_strength):
            mutation_type = random.randint(0, 11)
            
            if mutation_type < 3:
                section = random.choice(["encoder_layers", "decoder_layers"])
                if genome.hnn_architecture[section]:
                    idx = random.randint(0, len(genome.hnn_architecture[section]) - 1)
                    layer = genome.hnn_architecture[section][idx]
                    layer.type = random.choice(["linear", "conv", "conv_transpose", "dropout", "batch_norm", "max_pool", "avg_pool"])
                    layer.activation = random.choice(["GELU", "ReLU", "Sigmoid", "None", "LeakyReLU", "Tanh"])
                    layer.out_features = random.randint(32, 256)
                    layer.dropout_rate = random.uniform(0.1, 0.5) if layer.type == "dropout" else 0.0
                    if layer.type in ["conv", "conv_transpose", "max_pool", "avg_pool"]:
                        layer.kernel_size = random.choice([3, 5])
                        if layer.kernel_size % 2 == 0: layer.kernel_size = max(3, layer.kernel_size - 1)
                        layer.stride = random.choice([1, 2])

            elif mutation_type == 3:
                section = random.choice(["encoder_layers", "decoder_layers"])
                new_layer_def = LayerDefinition()
                genome.hnn_architecture[section].append(new_layer_def)

            elif mutation_type == 4:
                section = random.choice(["encoder_layers", "decoder_layers"])
                if len(genome.hnn_architecture[section]) > 1:
                    idx = random.randint(0, len(genome.hnn_architecture[section]) - 1)
                    del genome.hnn_architecture[section][idx]

            elif mutation_type == 5:
                genome.hnn_architecture["latent_dim_flat"] = max(16, min(512, genome.hnn_architecture["latent_dim_flat"] + random.randint(-16, 16)))
                for section_key in ["encoder_layers", "decoder_layers"]:
                    for layer in genome.hnn_architecture[section_key]:
                        if layer.type == "linear":
                            layer.out_features = max(16, layer.out_features + random.randint(-8, 8))
            
            elif mutation_type == 6:
                genome.learning_rate = max(1e-5, min(1e-2, genome.learning_rate * random.uniform(0.8, 1.2)))
            
            elif mutation_type == 7:
                genome.batch_size = random.choice([16, 32, 64, 128])

            elif mutation_type == 8:
                genome.training_epochs = random.randint(5, 20)
            
            elif mutation_type == 9:
                genome.qat_enabled = not genome.qat_enabled

            elif mutation_type == 10:
                genome.vram_limit_mb = max(128, min(4096, genome.vram_limit_mb + random.randint(-128, 128)))
            
            elif mutation_type == 11:
                section_bits = random.choice(["encoder", "decoder"])
                if genome.layer_bits.get(section_bits):
                    idx = random.randint(0, len(genome.layer_bits[section_bits]) - 1)
                    genome.layer_bits[section_bits][idx] = random.randint(1, 512)

            # Nauja: Evoliucionuoti savireguliacijos koeficientus
            genome.mutation_threshold += random.uniform(-genome.mutation_strength, genome.mutation_strength)
            genome.mutation_threshold = max(0.05, min(0.2, genome.mutation_threshold))
            genome.niche_radius += random.uniform(-genome.mutation_strength, genome.mutation_strength)
            genome.niche_radius = max(0.1, min(0.3, genome.niche_radius))
            genome.surrogate_usage += random.uniform(-genome.mutation_strength, genome.mutation_strength)
            genome.surrogate_usage = max(0.5, min(0.9, genome.surrogate_usage))
            genome.stagnation_threshold += random.choice([-1, 0, 1])
            genome.stagnation_threshold = max(2, min(5, genome.stagnation_threshold))
            genome.n_clusters += random.choice([-2, -1, 0, 1, 2])
            genome.n_clusters = max(3, min(10, genome.n_clusters))
            genome.mse_threshold += random.uniform(-genome.mutation_strength, genome.mutation_strength)
            genome.mse_threshold = max(0.3, min(0.7, genome.mse_threshold))
            genome.mutation_strength += random.uniform(-0.05, 0.05)
            genome.mutation_strength = max(0.05, min(0.2, genome.mutation_strength))

            # nD rutulio mutation
            r = genome.mutation_strength
            unit_vec = np.random.normal(0, 1, size=len(genome.params))
            unit_vec /= np.linalg.norm(unit_vec) if np.linalg.norm(unit_vec) != 0 else 1
            genome.params += r * unit_vec
            genome.params /= np.linalg.norm(genome.params) if np.linalg.norm(genome.params) != 0 else 1

        for section_key in ["encoder_layers", "decoder_layers"]:
            bits_section = section_key.replace("_layers", "")
            target_len = len(genome.hnn_architecture[section_key])
            current_len = len(genome.layer_bits.get(bits_section, []))
            if current_len < target_len:
                genome.layer_bits.setdefault(bits_section, []).extend([random.randint(1, 512) for _ in range(target_len - current_len)])
            elif current_len > target_len:
                genome.layer_bits[bits_section] = genome.layer_bits[bits_section][:target_len]
